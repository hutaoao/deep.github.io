## 一句话概括

闭包的实际价值不在"能访问外部变量"本身，而在于它能实现**私有变量**、**记忆状态**、**延迟执行**和**函数工厂**——防抖节流、模块封装、柯里化、React Hooks 等核心模式都依赖闭包实现。

## 背景与意义

### 为什么"知道闭包"不等于"会用闭包"？

很多开发者知道闭包的定义（函数 + 环境），但实际开发中却说不出哪里用到闭包。

```javascript
// ❌ 很多人的闭包"应用"局限在面试题
for (var i = 0; i < 5; i++) {
  setTimeout(() => console.log(i), 100);
}

// ✅ 实际项目中闭包无处不在：
useEffect(() => {
  // 闭包：回调函数捕获了 count
}, []);

const debouncedSearch = debounce(fetchResults, 300);
// 闭包：debounce 内部保存了 timer

const Counter = createStore({ count: 0 });
// 闭包：createStore 内部保存了 state
```

## 核心应用场景拆解

### 应用 1：防抖与节流（最高频）

**防抖（Debounce）—— 闭包保存定时器引用**

```javascript
// 核心：闭包保存 timer，每次调用都复用同一个 timer
function debounce(fn, delay) {
  let timer = null;  // ★ 闭包变量：保存定时器 ID

  return function (...args) {
    const context = this;

    if (timer) {
      clearTimeout(timer);  // 清除上次的定时器
    }

    timer = setTimeout(() => {
      fn.apply(context, args);
      timer = null;  // 执行完毕后重置
    }, delay);
  };
}

// 使用场景
const searchInput = document.getElementById('search');
const handleSearch = debounce((keyword) => {
  // 发送 API 请求
  console.log('搜索:', keyword);
}, 300);

searchInput.addEventListener('input', (e) => {
  handleSearch(e.target.value);
});
```

**节流（Throttle）—— 闭包保存状态标记**

```javascript
// 节流：每隔 delay 时间执行一次
function throttle(fn, delay) {
  let canRun = true;  // ★ 闭包变量

  return function (...args) {
    if (!canRun) return;  // 在冷却期，直接返回
    
    canRun = false;
    fn.apply(this, args);
    
    setTimeout(() => {
      canRun = true;  // 冷却期结束
    }, delay);
  };
}

// 使用场景：滚动事件、窗口 resize
window.addEventListener('scroll', throttle(() => {
  console.log('滚动位置:', window.scrollY);
}, 200));
```

### 应用 2：模块封装与私有变量

**立即执行函数（IIFE + 闭包）实现模块模式：**

```javascript
// 创建带私有变量的模块
const UserModule = (function() {
  // 私有状态（外部无法直接访问）
  let users = [];
  let idCounter = 0;

  // 私有方法（内部使用）
  function generateId() {
    return ++idCounter;
  }

  function log(action, user) {
    console.log(`[${new Date().toISOString()}] ${action}:`, user);
  }

  // 公开接口
  return {
    add(name) {
      const user = { id: generateId(), name, createdAt: new Date() };
      users.push(user);
      log('created', user);
      return user;
    },

    getById(id) {
      return users.find(u => u.id === id);
    },

    getAll() {
      return [...users];  // 返回拷贝，防止外部修改
    },

    remove(id) {
      const index = users.findIndex(u => u.id === id);
      if (index !== -1) {
        const removed = users.splice(index, 1);
        log('removed', removed[0]);
        return true;
      }
      return false;
    },

    get count() {
      return users.length;
    }
  };
})();

UserModule.add('张三');
UserModule.add('李四');
console.log(UserModule.getAll());  // [{ id: 1, name: '张三' }, { id: 2, name: '李四' }]
console.log(UserModule.count);     // 2
// console.log(UserModule.users);  // undefined ← 私有变量无法访问
```

**封装 Count 计数器：**

```javascript
function createCounter(initial = 0) {
  let count = initial;
  
  return {
    increment: () => ++count,
    decrement: () => --count,
    reset: () => { count = initial; },
    getCount: () => count
  };
}

const counter1 = createCounter(10);
const counter2 = createCounter(100);

console.log(counter1.increment());  // 11
console.log(counter2.increment());  // 101
// 每个实例有独立的作用域，互不干扰
```

### 应用 3：函数柯里化与偏函数

**柯里化（Currying）：**

```javascript
// 闭包逐层保存参数
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      // 参数够了，直接调用
      return fn(...args);
    } else {
      // 参数不够，返回新函数继续接收
      return function(...moreArgs) {
        return curried(...args, ...moreArgs);
      };
    }
  };
}

// 使用
const add = (a, b, c) => a + b + c;
const curriedAdd = curry(add);

console.log(curriedAdd(1)(2)(3));        // 6
console.log(curriedAdd(1, 2)(3));        // 6
console.log(curriedAdd(1, 2, 3));        // 6

// ★ 实际应用：日志函数预设参数
const log = (level, message, data) => {
  console.log(`[${level}] ${message}`, data || '');
};

const infoLog = curry(log)('INFO');
const errorLog = curry(log)('ERROR');

infoLog('用户登录', { userId: 42 });      // [INFO] 用户登录 { userId: 42 }
errorLog('数据库连接失败', { code: 500 }); // [ERROR] 数据库连接失败 { code: 500 }
```

### 应用 4：函数记忆（Memoization）

```javascript
// 闭包保存缓存
function memoize(fn) {
  const cache = new Map();  // ★ 闭包：保存计算结果

  return function (...args) {
    const key = JSON.stringify(args);
    
    if (cache.has(key)) {
      console.log('命中缓存:', key);
      return cache.get(key);
    }
    
    const result = fn(...args);
    cache.set(key, result);
    console.log('计算新值:', key);
    return result;
  };
}

// 使用：缓存昂贵的计算
const fibonacci = memoize((n) => {
  if (n <= 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);  // memorized 版本
});

console.log(fibonacci(40));  // 计算一次
console.log(fibonacci(40));  // 从缓存读取
```

### 应用 6：事件监听与回调中的闭包

```javascript
// ★ 闭包在事件绑定时"记住"上下文
function setupButtons() {
  const buttons = document.querySelectorAll('button');
  
  // 不使用闭包：所有按钮都输出最后一个索引
  for (var i = 0; i < buttons.length; i++) {
    buttons[i].onclick = function() {
      console.log(i);  // 始终输出 buttons.length
    };
  }

  // 使用闭包：每个按钮记住自己的索引
  for (var j = 0; j < buttons.length; j++) {
    (function(index) {
      buttons[index].onclick = function() {
        console.log(`按钮 ${index} 被点击`);
      };
    })(j);
  }
}
```

### 应用 7：定时器中的闭包

```javascript
function createTimer() {
  let count = 0;
  
  const timerId = setInterval(() => {
    console.log(++count);  // ★ 闭包捕获 count
    if (count >= 5) {
      clearInterval(timerId);
      console.log('计时结束');
    }
  }, 1000);

  return timerId;
}

// React Hooks 中的闭包陷阱
function TimerComponent() {
  const [count, setCount] = useState(0);
  
  useEffect(() => {
    const timer = setInterval(() => {
      console.log(count);  // ★ 闭包陷阱：捕获的是初始值 count=0
      setCount(count + 1);
    }, 1000);
    
    return () => clearInterval(timer);
  }, []);  // 依赖数组为空 → count 永远是 0
}
```

## 实战案例

### 案例一：完整防抖组件 Hook

```javascript
// React 中的防抖 Hook
function useDebounce(value, delay) {
  const [debouncedValue, setDebouncedValue] = useState(value);
  
  useEffect(() => {
    // ★ 闭包：定时器回调捕获了 setDebouncedValue
    const timer = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);
    
    return () => clearTimeout(timer);  // 清除旧的定时器
  }, [value, delay]);
  
  return debouncedValue;
}

// 使用
function SearchComponent() {
  const [query, setQuery] = useState('');
  const debouncedQuery = useDebounce(query, 300);
  
  useEffect(() => {
    if (debouncedQuery) {
      fetchResults(debouncedQuery);
    }
  }, [debouncedQuery]);
  
  return <input onChange={e => setQuery(e.target.value)} />;
}
```

// 只在第一次调用时初始化数据
function createLazyLoader() {

### 案例三：使用闭包跟踪异步操作

```javascript
function createAsyncTracker() {
  const pending = new Set();  // ★ 闭包

  return {
    start(id) {
      pending.add(id);
      console.log(`[${id}] 开始，待处理: ${pending.size}`);
    },
    
    complete(id) {
      pending.delete(id);
      console.log(`[${id}] 完成，待处理: ${pending.size}`);
      if (pending.size === 0) {
        console.log('所有请求已完成！');
      }
    },
    
    getPendingCount() {
      return pending.size;
    }
  };
}

// 使用
const tracker = createAsyncTracker();

[1, 2, 3].forEach(id => {
  tracker.start(id);
  fetch(`/api/data/${id}`)
    .then(() => tracker.complete(id))
    .catch(() => tracker.complete(id));
});
```

## 底层原理

### 闭包的内存回收时机

```javascript
function createClosure() {
  const data = { value: 'important' };
  const large = new Array(10000);  // 大对象
  
  return {
    getData: () => data,  // ★ 闭包引用 data
    // ★ large 没有被任何闭包引用 → V8 优化：不放入闭包上下文
    // getLarge: () => large  ← 如果加这行，large 也会被保留
  };
}

const closure = createClosure();
// 闭包上下文包含:
//   data: { value: 'important' } ← 保留
//   large: [...] ← ★ V8 优化：没有被引用，不保留

closure.getData = null;
// 当 getData 被置空后，data 也可以被 GC 回收了
```

### 多个闭包共享同一个作用域

```javascript
function createShared() {
  let counter = 0;  // ★ 同一个作用域

  return {
    increment() {
      counter++;  // 所有方法共享同一个 counter
    },
    decrement() {
      counter--;
    },
    getCount() {
      return counter;
    }
  };
}

const shared = createShared();
shared.increment();
shared.increment();
console.log(shared.getCount());  // 2
// increment、decrement、getCount 共享同一个 counter
```

## 高频面试题解析

### Q1：闭包在实际开发中有哪些应用场景？

**参考答案：**
1. **防抖/节流**：保存定时器引用
2. **模块模式**：实现私有变量和公开接口
3. **柯里化**：逐层保存参数
4. **函数记忆**：缓存计算结果
5. **单例模式**：保存实例引用
6. **事件绑定**：在循环中保存上下文
7. **React Hooks**：useEffect 的 cleanup 函数
8. **高阶函数**：withTimer、withLogger 等

### Q2：闭包和普通函数在内存上有什么不同？

**参考答案：**
闭包额外维护一个**Context 对象**（闭包作用域），存放被引用的外部变量。这个 Context 对象在闭包存活期间一直存在。普通函数的变量在执行完就被销毁了。

### Q3：闭包会导致内存泄漏吗？怎么避免？

**参考答案：**
闭包本身不是内存泄漏，但使用不当可能造成泄漏：
1. 闭包引用了大对象但没有清理
2. 闭包被全局变量引用导致无法释放
3. 事件监听器绑定的闭包未被移除

**避免方案**：
- 只保留需要的变量
- 使用完置空
- 及时移除事件监听
- 组件销毁时清理闭包引用

### Q4：React Hooks 中的闭包陷阱是什么？

**参考答案：**
使用 useEffect 等 Hook 时，回调函数以闭包形式捕获了 render 时的变量值。如果依赖数组不包含这个变量，回调中访问的永远是旧值。

```javascript
// 闭包陷阱
const [count, setCount] = useState(0);

useEffect(() => {
  const timer = setInterval(() => {
    console.log(count);  // 永远是 0
    setCount(count + 1); // 永远设置成 1
  }, 1000);
  return () => clearInterval(timer);
}, []);
```

**解决方案**：用函数式更新 `setCount(c => c + 1)` 或正确设置依赖。
